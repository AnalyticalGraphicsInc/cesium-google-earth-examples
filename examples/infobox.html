Cesium Examples - HTML String Balloons

Relevant Cesium Source

var infoBoxContainer = document.createElement('div');
infoBoxContainer.className = 'cesium-viewer-infoBoxContainer';
viewer.container.appendChild(infoBoxContainer);
var infoBox = new Cesium.InfoBox(infoBoxContainer);
var infoBoxViewModel = infoBox.viewModel;
infoBoxViewModel.titleText = 'Cesium Logo';
infoBoxViewModel.descriptionRawHtml = '<img alt="CESIUMJS" src="./static/Cesium_Logo_Color.jpg"/>';
infoBoxViewModel.showInfo = true;

Relevant Google Earth Source

var balloon = ge.createHtmlDivBalloon('');
balloon.setFeature(placemark); // optional
balloon.setMaxWidth(800);

// create the <div>
var div = document.createElement('DIV');
div.innerHTML =
        '<img src="http://www.google.com/googlegulp/images/logo.gif"><br>'
        + '<a href="http://www.google.com/googlegulp/">Google Gulp</a>';
balloon.setContentDiv(div);

ge.setBalloon(balloon);
